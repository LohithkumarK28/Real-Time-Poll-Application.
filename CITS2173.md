const express = require("express");

const app = express();
const server = http.createServer(app);
const io = new Server(server);

const PORT = process.env.PORT || 3000;

app.use(express.json());
app.use(express.static(path.join(__dirname, "public")));

      { id: 2, text: "Backend", votes: 11 },
      { id: 3, text: "Full Stack", votes: 22 },
      { id: 4, text: "UI/UX Design", votes: 8 }
    ]
  }

app.get("/api/polls", (req, res) => {
  res.json(polls);
});

app.post("/api/polls", (req, res) => {
  const { question, createdBy, options } = req.body;

  if (
    !question ||
    !Array.isArray(options) ||
    options.length < 2
  ) {
    return res.status(400).json({
      message: "A question and at least two options are required."
    });
  }

  const cleanOptions = options
    .map(option => String(option).trim())
    .filter(Boolean);

  if (cleanOptions.length < 2) {
    return res.status(400).json({
      message: "Please provide at least two valid options."
    });
  }

  const newPoll = {
    id: Date.now(),
    question: question.trim(),
    createdBy:
      createdBy?.trim() || "Anonymous",
    options: cleanOptions.map((text, index) => ({
      id: index + 1,
      text,
      votes: 0
    }))
  };

  polls.unshift(newPoll);

  io.emit("pollsUpdated", polls);

  res.status(201).json(newPoll);
});

app.post("/api/polls/:pollId/vote", (req, res) => {
  const pollId = Number(req.params.pollId);
  const optionId = Number(req.body.optionId);

  const poll = polls.find(
    item => item.id === pollId
  );

  if (!poll) {
    return res.status(404).json({
      message: "Poll not found."
    });
  }

  const option = poll.options.find(
    item => item.id === optionId
  );

  if (!option) {
    return res.status(404).json({
      message: "Option not found."
    });
  }

  option.votes += 1;

  io.emit("pollUpdated", poll);

  res.json(poll);
});

io.on("connection", socket => {
  console.log(
    "A user connected:",
    socket.id
  );

  socket.on("disconnect", () => {
    console.log(
      "A user disconnected:",
      socket.id
    );
  });
});

server.listen(PORT, () => {
  console.log(
    `Poll application running on http://localhost:${PORT}`
  );
});

const socket = io();

let polls = [];
let votedPolls =
  JSON.parse(
    localStorage.getItem("votedPolls")
  ) || [];

async function loadPolls() {
  try {
    const response =
      await fetch("/api/polls");

    polls = await response.json();

    displayPolls();
  } catch (error) {
    console.error(
      "Unable to load polls:",
      error
    );
  }
}

function displayPolls() {
  const container =
    document.getElementById(
      "poll-container"
    );

  container.innerHTML = "";

  polls.forEach(poll => {
    const card =
      document.createElement("div");

    card.className = "poll-card";

    const hasVoted =
      votedPolls.includes(poll.id);

    const totalVotes =
      poll.options.reduce(
        (total, option) =>
          total + option.votes,
        0
      );

    let optionsHTML = "";

    poll.options.forEach(option => {
      if (hasVoted) {
        const percentage =
          totalVotes === 0
            ? 0
            : Math.round(
                (option.votes /
                  totalVotes) *
                  100
              );

        optionsHTML += `
          <div class="option">

            <div class="result-row">

              <span>
                ${escapeHtml(option.text)}
              </span>

              <strong>
                ${percentage}%
              </strong>

            </div>

            <div class="progress">

              <div
                class="progress-fill"
                style="width:${percentage}%"
              ></div>

            </div>

          </div>
        `;
      } else {
        optionsHTML += `
          <div class="option">

            <button
              class="option-button"
              onclick="vote(
                ${poll.id},
                ${option.id}
              )"
            >
              ${escapeHtml(option.text)}
            </button>

          </div>
        `;
      }
    });

    card.innerHTML = `
      <span class="poll-status">
        ● LIVE
      </span>

      <h2>
        ${escapeHtml(poll.question)}
      </h2>

      <p class="poll-author">
        Created by
        ${escapeHtml(poll.createdBy)}
      </p>

      ${optionsHTML}

      <p class="total-votes">
        ${totalVotes} total votes
      </p>
    `;

    container.appendChild(card);
  });
}

async function vote(
  pollId,
  optionId
) {
  if (
    votedPolls.includes(pollId)
  ) {
    return;
  }

  try {
    const response =
      await fetch(
        `/api/polls/${pollId}/vote`,
        {
          method: "POST",

          headers: {
            "Content-Type":
              "application/json"
          },

          body: JSON.stringify({
            optionId
          })
        }
      );

    if (!response.ok) {
      throw new Error(
        "Vote failed"
      );
    }

    votedPolls.push(pollId);

    localStorage.setItem(
      "votedPolls",
      JSON.stringify(votedPolls)
    );

    showToast();

    await loadPolls();
  } catch (error) {
    alert(
      "Unable to submit your vote."
    );
  }
}

function showToast() {
  const toast =
    document.getElementById("toast");

  toast.classList.remove(
    "hidden"
  );

  setTimeout(() => {
    toast.classList.add(
      "hidden"
    );
  }, 2500);
}

function addOption() {
  const list =
    document.getElementById(
      "options-list"
    );

  const currentOptions =
    document.querySelectorAll(
      ".poll-option"
    );

  if (
    currentOptions.length >= 6
  ) {
    alert(
      "Maximum 6 options allowed."
    );

    return;
  }

  const input =
    document.createElement(
      "input"
    );

  input.className =
    "poll-option";

  input.placeholder =
    `Option ${
      currentOptions.length + 1
    }`;

  list.appendChild(input);
}

document
  .getElementById("poll-form")
  .addEventListener(
    "submit",
    async event => {
      event.preventDefault();

      const question =
        document
          .getElementById("question")
          .value
          .trim();

      const createdBy =
        document
          .getElementById("creator")
          .value
          .trim();

      const options =
        Array.from(
          document.querySelectorAll(
            ".poll-option"
          )
        )
          .map(input =>
            input.value.trim()
          )
          .filter(Boolean);

      if (options.length < 2) {
        alert(
          "Please enter at least two options."
        );

        return;
      }

      try {
        const response =
          await fetch(
            "/api/polls",
            {
              method: "POST",

              headers: {
                "Content-Type":
                  "application/json"
              },

              body: JSON.stringify({
                question,
                createdBy,
                options
              })
            }
          );

        if (!response.ok) {
          throw new Error(
            "Unable to create poll"
          );
        }

        alert(
          "Your poll is now live!"
        );

        event.target.reset();

        resetOptions();

        showPolls();

        await loadPolls();
      } catch (error) {
        alert(
          "Unable to create the poll."
        );
      }
    }
  );

    >
  `;
}

function escapeHtml(text) {
  const div =
    document.createElement("div");

  div.textContent = text;

  return div.innerHTML;
}

socket.on(
  "pollUpdated",
  updatedPoll => {
    const index =
      polls.findIndex(
        poll =>
          poll.id ===
          updatedPoll.id
      );

    if (index !== -1) {
      polls[index] =
        updatedPoll;

      displayPolls();
    }
  }
);

socket.on(

  updatedPolls => {
    polls = updatedPolls;

    displayPolls();

);
